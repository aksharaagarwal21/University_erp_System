6.2 Database Connectivity (Back-End Server)
-------------------------------------------

The back-end server connects to MySQL using the mysql2/promise library with a
connection pool for efficient database access.

File: backend/server.js

// ---- Database Connection Setup ----
const express = require('express');
const cors = require('cors');
const mysql = require('mysql2/promise');

const app = express();
app.use(cors());
app.use(express.json());

// MySQL Connection Pool
const pool = mysql.createPool({
  host: 'localhost',
  user: 'root',
  password: '****',
  database: 'university_erpsystem',
  waitForConnections: true,
  connectionLimit: 10
});

pool.getConnection()
  .then(c => { console.log('MySQL Connected'); c.release(); })
  .catch(e => console.error('MySQL Error:', e.message));

Explanation:
- A connection pool with 10 max connections is created to handle concurrent
  requests efficiently.
- The pool.getConnection() method verifies the connection on server startup.
- All subsequent queries use pool.query() which automatically acquires and
  releases connections from the pool.

-------------------------

The system implements role-based authentication with three user types:
Student, Faculty, and Super Admin.

// ---- Users Table (SQL) ----
CREATE TABLE Users (
    UserID INT AUTO_INCREMENT PRIMARY KEY,
    Username VARCHAR(50) UNIQUE NOT NULL,
    Password VARCHAR(255) NOT NULL,
    Role ENUM('student', 'faculty', 'admin') NOT NULL,
    RefID INT DEFAULT NULL,
    FullName VARCHAR(100) NOT NULL,
    Email VARCHAR(100),
    CreatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

// ---- Login API Endpoint ----
app.post('/api/auth/login', async (req, res) => {
  try {
    const { username, password, role } = req.body;
    const [users] = await pool.query(
      'SELECT * FROM Users WHERE Username=? AND Password=? AND Role=?',
      [username, password, role]
    );
    if (users.length === 0)
      return res.status(401).json({ error: 'Invalid credentials' });
    const user = users[0];
    res.json({
      UserID: user.UserID,
      Username: user.Username,
      Role: user.Role,
      RefID: user.RefID,
      FullName: user.FullName
    });
  } catch (err) {
    res.status(500).json({ error: err.message });
  }
});

// ---- Registration API Endpoint ----
app.post('/api/auth/register', async (req, res) => {
  try {
    const { Username, Password, Role, RefID, FullName, Email } = req.body;
    await pool.query(
      'INSERT INTO Users (Username,Password,Role,RefID,FullName,Email) VALUES (?,?,?,?,?,?)',
      [Username, Password, Role, RefID || null, FullName, Email]
    );
    res.json({ message: 'Registration successful' });
  } catch (err) {
    res.status(500).json({ error: err.message });
  }
});

6.4 CRUD API Implementation
----------------------------

A generic CRUD helper function is used to generate REST endpoints for all
9 database tables, reducing code duplication:

function crud(endpoint, table, idField) {
  // GET all records
  app.get(`/api/${endpoint}`, async (req, res) => {
    const [r] = await pool.query(`SELECT * FROM ${table} ORDER BY ${idField}`);
    res.json(r);
  });

  // POST - Create new record
  app.post(`/api/${endpoint}`, async (req, res) => {
    const keys = Object.keys(req.body);
    const vals = Object.values(req.body);
    await pool.query(
      `INSERT INTO ${table} (${keys.join(',')}) VALUES (${keys.map(()=>'?').join(',')})`,
      vals
    );
    res.json({ message: `${table} added successfully` });
  });

  // PUT - Update record
  app.put(`/api/${endpoint}/:id`, async (req, res) => {
    const sets = Object.keys(req.body).filter(k => k !== idField)
                       .map(k => `${k}=?`).join(',');
    const vals = Object.keys(req.body).filter(k => k !== idField)
                       .map(k => req.body[k]);
    await pool.query(`UPDATE ${table} SET ${sets} WHERE ${idField}=?`,
                     [...vals, req.params.id]);
    res.json({ message: `${table} updated successfully` });
  });

  // DELETE - Remove record
  app.delete(`/api/${endpoint}/:id`, async (req, res) => {
    await pool.query(`DELETE FROM ${table} WHERE ${idField}=?`, [req.params.id]);
    res.json({ message: `${table} deleted` });
  });
}

------------------------------------

These endpoints return data filtered by a specific student's ID, used by the
Student Dashboard:

// Student Profile with Department Name
app.get('/api/student/:id/profile', async (req, res) => {
  const [r] = await pool.query(
    'SELECT s.*, d.DeptName FROM Student s ' +
    'LEFT JOIN Department d ON s.DeptID = d.DeptID ' +
    'WHERE s.StudentID = ?', [req.params.id]
  );
  res.json(r[0] || {});
});

// Student's Enrolled Subjects
app.get('/api/student/:id/enrollments', async (req, res) => {
  const [r] = await pool.query(
    'SELECT e.*, sub.SubjectName, sub.Credits ' +
    'FROM Enrollment e JOIN Subject sub ON e.SubjectID = sub.SubjectID ' +
    'WHERE e.StudentID = ?', [req.params.id]
  );
  res.json(r);
});

// Student's Exam Results
app.get('/api/student/:id/results', async (req, res) => {
  const [r] = await pool.query(
    'SELECT r.*, e.ExamDate, sub.SubjectName FROM Result r ' +
    'JOIN Exam e ON r.ExamID = e.ExamID ' +
    'JOIN Subject sub ON e.SubjectID = sub.SubjectID ' +
    'WHERE r.StudentID = ?', [req.params.id]
  );
  res.json(r);
});

// Student's Timetable
app.get('/api/student/:id/timetable', async (req, res) => {
  const [r] = await pool.query(
    'SELECT t.Day, t.TimeSlot, t.RoomNo, f.FacultyName, sub.SubjectName ' +
    'FROM Enrollment en JOIN Subject sub ON en.SubjectID = sub.SubjectID ' +
    'JOIN Timetable t ON sub.SubjectID = t.SubjectID ' +
    'JOIN Faculty f ON t.FacultyID = f.FacultyID ' +
    'WHERE en.StudentID = ? ' +
    "ORDER BY FIELD(t.Day,'Monday','Tuesday','Wednesday','Thursday','Friday'), t.TimeSlot",
    [req.params.id]
  );
  res.json(r);
});

app.post('/api/query', async (req, res) => {
  try {
    const { sql } = req.body;
    if (!sql.trim().toUpperCase().startsWith('SELECT'))
      return res.status(403).json({ error: 'Only SELECT queries allowed' });
    const [rows] = await pool.query(sql);
    res.json({
      columns: rows.length > 0 ? Object.keys(rows[0]) : [],
      data: rows
    });
  } catch (err) {
    res.status(500).json({ error: err.message });
  }
});

6.7.1 Application Entry Point (App.jsx)

The main App component handles authentication state, role-based routing,
and the sidebar navigation:

function App() {
  const [user, setUser] = useState(() => {
    const saved = localStorage.getItem('erp_user');
    return saved ? JSON.parse(saved) : null;
  });

  const handleLogin = (userData) => {
    setUser(userData);
    localStorage.setItem('erp_user', JSON.stringify(userData));
  };

  const handleLogout = () => {
    setUser(null);
    localStorage.removeItem('erp_user');
  };

  // If not logged in, show login page
  if (!user) return <LoginPage onLogin={handleLogin} />;

  // Role-based dashboard routing
  return (
    <BrowserRouter>
      <div className="app-layout">
        <Sidebar user={user} onLogout={handleLogout} />
        <main className="main-content">
          {user.Role === 'student' && <StudentDashboard user={user} />}
          {user.Role === 'faculty' && <FacultyDashboard user={user} />}
          {user.Role === 'admin' && (
            <Routes>
              <Route path="/" element={<Dashboard />} />
              <Route path="/students" element={<Students />} />
              <Route path="/faculty" element={<Faculty />} />
              {/* ... other admin routes ... */}
            </Routes>
          )}
        </main>
      </div>
    </BrowserRouter>
  );
}

7.3 Discussion
--------------

7.3.1 System Architecture

The University ERP System follows a three-tier architecture:

Tier 1 (Presentation Layer): React.js front-end served via Vite dev server
on port 5173. Provides role-based dashboards with interactive charts,
CRUD forms, and real-time data visualization.

Tier 2 (Application Layer): Node.js + Express back-end running on port 5000.
Handles authentication, REST API routing, business logic, and SQL query
execution.

Tier 3 (Data Layer): MySQL 8.0 database containing normalized tables,
views, triggers, and stored procedures implementing all DBMS concepts
from Chapters 1-5.


7.3.2 Key Achievements

1. Complete Database Implementation: All 9 tables with proper primary keys,
   foreign keys, and referential integrity constraints were implemented as
   designed in the ER model (Chapter 1).

2. Complex Query Support: The SQL Console allows execution of all query types
   demonstrated in Chapter 3 including aggregate functions, joins, subqueries,
   views, and set operations.

3. Normalization Verified: The database schema satisfies all normal forms
   from 1NF through 5NF as analyzed in Chapter 4.

4. Concurrency Control: The transaction mechanisms from Chapter 5 are
   demonstrated through the marks management feature where multiple faculty
   members can safely update student results.

5. Trigger Automation: The Assign_Grade trigger automatically calculates
   grades when marks are inserted, reducing manual effort and ensuring
   consistency.

6. Role-Based Access Control: Three distinct user interfaces ensure that
   students can only view their own data, faculty can manage their subject
   results, and administrators have full system control.

7. Data Visualization: Interactive charts (bar, pie, radial) provide
   immediate visual insights into enrollment patterns, grade distributions,
   and department performance.


7.3.3 Limitations and Future Scope

Limitations:
- Passwords are stored in plaintext (should use bcrypt hashing in production)
- No JWT token-based session management
- Limited to SELECT queries in SQL Console for safety

Future Scope:
- Add attendance tracking module
- Implement email notifications for result publication
- Add PDF report generation for student transcripts
- Integrate with university payment systems for fee management
- Deploy on cloud infrastructure (AWS/Azure) for scalability
